building_sm_conduction_nexus = {
	base_buildtime = @b2_time
	category = resource
	empire_limit = {
		base = 1
	}

	allow = {
		has_upgraded_capital = yes
	}
	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			AND = {
				exists = owner
				owner = {
					is_ai = yes
				}
				has_any_generator_district_or_building = no
			}
		}
	}

	potential = {
		NOR = {
			has_modifier = resort_colony
		}
		OR = {
			NOT = { is_planet_class = pc_habitat }
			AND = {
				is_planet_class = pc_habitat
				check_modifier_value = {
					modifier = district_hab_energy_max
					value > 0
				}
			}
		}
		OR = {
			owner = {
				is_ai = no
			}
			has_any_generator_district_or_building = yes
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			exotic_gases = @b2_rare_cost
		}
		upkeep = {
			energy = 2
			exotic_gases = @b2_rare_upkeep
		}
	}

	planet_modifier = {
		planet_technician_physics_research_produces_add = 2
	}

	inline_script = {
		script = jobs/technicians_add
		AMOUNT = 1
	}

	prerequisites = {
		tech_sm_conduction_nexus
	}
}
