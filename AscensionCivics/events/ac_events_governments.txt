namespace = ascension_civics
# On genetics tradition finsihed
country_event = {
	id = ascension_civics.1000
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = {
			has_country_flag = ac_has_finished_genetics_tradition
		}
		has_tradition = tr_genetics_finish
	}
	immediate = {
		# Makes sure this is only triggered once
		set_country_flag = ac_has_finished_genetics_tradition
		# Starts situation
	}
}

leader_event = {
	id = ascension_civics.1100
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_authority = auth_imperial
		has_tradition = tr_prosperity_adopt
		# has_country_flag = ac_genetic_perfect
	}
	immediate = {
		owner = {
			ruler = {
				save_event_target_as = ruler_to_be_cloned
			}
			clone_leader = {
				target = ruler_to_be_cloned
				set_age = 1
				effect = {
					add_trait_no_notify = trait_imperial_heir
				}
			}
		}
		
	}
}
