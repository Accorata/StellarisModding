namespace = ascension_civics
@necessary_avatar_points = 12 # Once a year
# On post government change
country_event = {
	id = ascension_civics.200
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_valid_civic = civic_doomseekers
	}
	immediate = {
		if = {
			limit = {
				NOT = {
					is_variable_set = ac_avatars_summoned
				}
			}
			set_variable = {
				which = ac_avatars_summoned
				value = 0
			}
		}
		set_variable = {
			which = ac_avatar_progress
			value = 0
		}
	}
}

# On monthly pulse
country_event = {
	id = ascension_civics.201
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_valid_civic = civic_doomseekers
	}
	immediate = {
		change_variable = {
			which = ac_avatar_progress
			value = 1
		}
		if = {
			limit = {
				has_tech = "tech_ac_doom_foretold"
			}
			change_variable = {
				which = ac_avatar_progress
				value = 2
			}
		}
		every_owned_starbase = {
			limit = {
				has_starbase_building = ac_avatar_caller
			}
			ROOT = {
				change_variable = {
					which = ac_avatar_progress
					value = 1
				}
			}
		}
		if = {
			limit = {
				check_variable_arithmetic = {
					which = ac_avatar_progress
					value >= @necessary_avatar_points
				}
			}
			country_event = {
				id = ascension_civics.202
			}
			subtract_variable = {
				which = ac_avatar_progress
				value = @necessary_avatar_points
			}
			change_variable = {
				which = ac_avatars_summoned
				value = 1
			}
		}
	}
}

# From monthly pulse counting avatar progress - summon an avatar
country_event = {
	id = ascension_civics.202
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = {
				NOT = {
					exists = event_target:shroud_country
				}
			}
			create_country = {
				name = "NAME_Creatures_of_the_Shroud"
				type = shroud_spirits
				flag = {
					icon = {
						category = "special"
						file = "the_shroud.dds"
					}
					background = {
						category = "backgrounds"
						file = "00_solid.dds"
					}
					colors = {
						"dark_purple"
						"black"
						"null"
						"null"
					}
				}
			}
			last_created_country = {
				establish_communications_no_message = root
				save_global_event_target_as = shroud_country
			}
		}
		random_system_within_border = {
			save_event_target_as = shroud_spawn_system
			random_system_planet = {
				save_event_target_as = shroud_spawn_planet
			}
		}
		random_system_within_border = {
			limit = {
				starbase = {
					has_starbase_building = ac_avatar_caller
				}
			}
			save_event_target_as = shroud_spawn_system
			random_system_planet = {
				save_event_target_as = shroud_spawn_planet
			}
		}
		create_fleet = {
			name = "NAME_Shroud_Manifestation"			#"NAME_Corrupted_Avatar"
			settings = {
				spawn_debris = no
				is_boss = yes
			}
			effect = {
				set_owner = event_target:shroud_country
				# create_ship = {
				# 	name = "NAME_Corrupted_Avatar"
				# 	design = "NAME_Corrupted_Avatar"
				# }
				create_ship = {
					name = "NAME_Fragment"
					design = "NAME_Shroud_Manifestation"
				}
				set_location = event_target:shroud_spawn_planet
				set_fleet_stance = neutral
			}
		}
		#?
		create_message = {
			type = MESSAGE_SHROUD_BREACH
			localization = MESSAGE_SHROUD_BREACH_DESC
			days = 1
			target = root
			variable = {
				type = name
				localization = COUNTRY
				scope = root
			}
		}
	}
}

	is_triggered_only = yes
	hide_window = yes
	trigger = {
		has_valid_civic = civic_doomseekers
		exists = event_target:shroud_country

	}
	immediate = {
		every_country = {
			limit = {
				country = event_target:shroud_country
			}
			country_event = {
				id = ascension_civics.211
			}
		}
	}
}
