# This = target country (player)
# From = source country

		cw_enclave.100		# Criminal Enclave greetings
		# criminal_worlds.52		# White Flower Hostage
	}

### Traditions ###

	events = {
		criminal_worlds.107		# Pirates become neutral to Pirate Warlords (criminality tradition)
	}
