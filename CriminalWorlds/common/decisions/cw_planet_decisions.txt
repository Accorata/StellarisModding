decision_uplift_slums = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_arcology_project
	enactment_time = 30
	resources = {
		category = decisions
		cost = {
			unity = 250
		}
	}
	potential = {
		exists = owner
		OR = {
			has_modifier = cw_underground_slums_1
			has_modifier = cw_underground_slums_2
		}
	}
	allow = {
		free_housing > 20
	}
	effect = {
		if = {
			limit = {
				has_modifier = cw_underground_slums_1
			}
			remove_modifier = cw_underground_slums_1
		}
		if = {
			limit = {
				has_modifier = cw_underground_slums_2
			}
			remove_modifier = cw_underground_slums_2
		}
	}
	ai_weight = {
		weight = 100
	}
}

decision_cw_undercity_expansion = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_arcology_project
	enactment_time = 180
	resources = {
		category = decisions
		cost = {
			minerals = 500
			alloys = 100
			influence = 10
		}
	}
	potential = {
		exists = owner
		OR = {
			is_planet_class = pc_city
			has_modifier = cw_ruined_undercity
			# For possible modders
			uses_district_set = city_world
		}
	}
	allow = {
		count_deposits = {
			category = d_cw_lower_level
			count < 5
		}
	}
	effect = {
		add_deposit = d_cw_lower_level
	}
	ai_weight = {
		weight = 0
		modifier = {
			add = 100
			num_free_districts = 0
		}
	}
}
