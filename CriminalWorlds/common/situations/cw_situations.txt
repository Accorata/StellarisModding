### Upgrade Enclave Situation
situation_cw_enclave_upgrade = {
	picture = GFX_evt_dark_alley
	complete_icon_frame = GFX_situation_outcome_frame_green
	complete_icon = GFX_situation_outcome_unknown
	on_progress_complete = {
		custom_tooltip = situation_cw_enclave_upgrade_complete
		# Upgrades enclave
		situation_event = {
			id = cw_enclave.4
		}
	}
	abort_trigger = {
		NOT = {
			exists = root.owner
		}
	}
	stages = {
		stage_1 = {
			icon = GFX_situation_stage_1
			icon_background = GFX_situation_stage_frame_blue
			end = 100
		}
	}
	monthly_progress = {
		# The situation takes 100 months without interference
		base = 1
		modifier = {
			mult = 3
			desc = cw_recent_deal_three_times			# To prevent error log message
			owner = {
				has_country_flag = cw_recent_deal
			}
		}
	}
	approach = {
		name = situation_cw_enclave_upgrade_approach
		icon = GFX_situation_approach_this_is_fine
		icon_background = GFX_situation_approach_bg_yellow
		default = yes
		on_select = {
			custom_tooltip = deficit_approach_do_nothing_tooltip
		}
	}
}
