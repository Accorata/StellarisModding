# Auth Scripted Triggers
trait_ac_has_edict_unholy_possession = {
	OR = {
		AND = {
			exists = owner
			owner = {
				has_edict = ac_unholy_possession
			}
		}
		AND = {
			exists = from
			from = {
				is_scope_type = country
				has_edict = ac_unholy_possession
			}
		}
	}
}
